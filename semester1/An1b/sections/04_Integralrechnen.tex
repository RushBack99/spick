\subsection{Obersumme / Untersumme}

Die Fläche unter einer Funktion $f(x)$ wird in Intervalle zerlegt.

\textbf{Voraussetzung: $f: [a;b] \to \mathbb{R}$ und $\mathbb{W}_f$ beschränkt} 

\renewcommand{\arraystretch}{1.5}
\begin{tabular}{@{}ll@{}}
    Zerlegung                           & $Z = \lbrace x_0; x_1; \ldots ; x_n \rbrace = \lbrace x_i$  $\vert i=0; \ldots ; n \rbrace$ \quad $(n \in \mathbb{N}; n \geq 2)$ \\
    Breite(n) der Intervalle            & $\Delta x_i = x_{i+1} - x_i$ \quad $\Delta x$ kann variieren! \\
    Maschenweite                        & $\mathrm{max}(\Delta x_i) = d(Z)$ \\
    \textbf{Äquidistante Zerlegung}     & $Z = \lbrace a + \frac{b-a}{n} \cdot i $ $\vert i=0; \ldots; n \rbrace$
\end{tabular}

\vspace{0.4cm}

\begin{minipage}[c]{0.48\columnwidth}
    \textbf{Untersumme \myul{US}} 
    $$ \rm US = \sum \limits_{i=0}^{n-1} m_i \cdot \Delta x_i $$
    $$ m_i = \mathrm{inf} \lbrace f(x) \, \vert x_i \leq x \leq x_{i+1} \rbrace $$

    \textbf{Obersumme $\overline{\rm OS}$} 
    $$ \rm OS = \sum \limits_{i=0}^{n-1} M_i \cdot \Delta x_i $$
    $$ M_i = \mathrm{sup} \lbrace f(x) \, \vert \,  x_i \leq x \leq x_{i+1} \rbrace $$

\subsection{Flächen unter Integralen}

\textbf{Voraussetzung: $f: [a;b] \to \mathbb{R}$ und $\mathbb{W}_f$ beschränkt}
$$ \text{Fläche:} \quad A = \int \limits_{a}^{b} \abs{ f(x) } \, \diff x $$
\textcolor{red}{Der Inhalt des Betrags muss auf Vorzeichen untersucht werden!} \\
Negative Vorzeichen müssen über die x-Achse geklappt werden: \\
$\vert x^2 - x \vert = -(x^2 - x)$ falls $x > x^2$
